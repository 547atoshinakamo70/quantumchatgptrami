# wormhole_predictor.py
from __future__ import annotations
from dataclasses import dataclass
from typing import Dict, List, Tuple
import math, numpy as np

# --- Qiskit wormhole ---
from qiskit import QuantumCircuit, Aer, execute

CATEGORIES = [
    "latency", "retrieval", "security", "wallet",
    "mining", "zk", "coinjoin", "ui"
]

@dataclass
class Signals:
    # Rellena estos con tus m√©tricas reales
    trend_strength: float     # 0..1 (adopci√≥n/crecimiento)
    volatility: float         # 0..1 (ruido/regresiones)
    dev_velocity: float       # 0..1 (cadencia de entrega)
    user_frustration: float   # 0..1 (errores, ca√≠das, quejas)
    infra_cost: float         # 0..1 (coste presi√≥n)
    security_findings: float  # 0..1 (hallazgos recientes)

def _clip01(x: float) -> float:
    return max(0.0, min(1.0, float(x)))

def build_wormhole_circuit(sig: Signals) -> QuantumCircuit:
    """
    Codifica se√±ales en rotaciones, aplica SWAP (wormhole) y mide.
    3 qubits ‚Üí 3 bits ‚Üí luego expandimos a 8 categor√≠as.
    """
    qc = QuantumCircuit(3, 3)
    # Encode signals in rotations (escala a [0, 2œÄ])
    angles = [
        2*math.pi*_clip01(sig.trend_strength + 0.15*sig.dev_velocity),
        2*math.pi*_clip01(0.5*sig.volatility + 0.5*sig.user_frustration),
        2*math.pi*_clip01(0.6*sig.security_findings + 0.4*sig.infra_cost),
    ]
    for i, a in enumerate(angles):
        qc.ry(a, i)
        qc.rz(a/2, i)

    # ‚ÄúWormhole‚Äù: mezcla y cruza informaci√≥n temporal/causal
    qc.h(0)
    qc.x(1)
    qc.swap(0, 1)
    qc.cx(1, 2)
    qc.h(0)

    # Medida
    qc.measure([0,1,2], [0,1,2])
    return qc

def sample_weights_from_circuit(qc: QuantumCircuit, shots: int = 2048) -> Dict[str, float]:
    backend = Aer.get_backend('qasm_simulator')
    result = execute(qc, backend, shots=shots).result()
    counts = result.get_counts()

    # 3 bits ‚Üí 8 estados ‚Üí mapea directamente a 8 categor√≠as
    total = sum(counts.values())
    probs8 = np.zeros(8, dtype=float)
    for bitstring, c in counts.items():
        # bitstring como '101' ‚Üí √≠ndice
        idx = int(bitstring, 2)
        probs8[idx] = c / total

    # Normaliza y suaviza
    probs8 = (probs8 + 1e-6) / (probs8.sum() + 1e-6 * len(probs8))

    weights = {cat: float(probs8[i]) for i, cat in enumerate(CATEGORIES)}
    return weights

# --- Heur√≠stica / reglas (puedes cambiar) ---
SUGGESTIONS = {
    "latency":  "Activar streaming + cache de logits + batching; medir p95.",
    "retrieval":"RAG con √≠ndice vectorial (FAISS), refresh incremental y guardado por dominios.",
    "security": "L√≠mites/quotas, saneamiento entradas, firmas en RPC, auditor√≠a y fuzz.",
    "wallet":   "Sincron√≠a robusta wallet<->chain, gesti√≥n claves y 2FA opcional.",
    "mining":   "Planificador de miner√≠a v√≠a HTTP con colas y backoff; m√©tricas de shares.",
    "zk":       "Precompilar circuitos ZK, cach√© de pruebas, APIs asincr√≥nicas.",
    "coinjoin": "Mejorar selector de monedas + coinjoin programable con pol√≠ticas.",
    "ui":       "Onboarding simple, plantillas de prompts y diagn√≥sticos visibles."
}

def rank_improvements(weights: Dict[str, float], sig: Signals, top_k: int = 6) -> List[Tuple[str, float, str]]:
    """
    Combina pesos cu√°nticos + una priorizaci√≥n simple seg√∫n se√±ales.
    """
    # Boosts por situaci√≥n:
    boosts = {
        "security": 1.0 + 0.7*_clip01(sig.security_findings),
        "latency":  1.0 + 0.6*_clip01(sig.user_frustration),
        "retrieval":1.0 + 0.4*_clip01(sig.trend_strength),
        "ui":       1.0 + 0.3*_clip01(sig.trend_strength),
        "mining":   1.0 + 0.4*_clip01(sig.dev_velocity),
        "zk":       1.0 + 0.4*_clip01(sig.dev_velocity),
        "coinjoin": 1.0 + 0.3*_clip01(sig.volatility),
        "wallet":   1.0 + 0.4*_clip01(sig.user_frustration),
    }

    scored = []
    for cat in CATEGORIES:
        base = weights.get(cat, 0.0)
        score = base * boosts.get(cat, 1.0)
        scored.append((cat, score, SUGGESTIONS[cat]))

    scored.sort(key=lambda x: x[1], reverse=True)
    return scored[:top_k]

# --- API principal que usaremos desde Flask ---
def predict_roadmap(signals_dict: Dict[str, float]) -> Dict:
    sig = Signals(
        trend_strength=signals_dict.get("trend_strength", 0.5),
        volatility=signals_dict.get("volatility", 0.5),
        dev_velocity=signals_dict.get("dev_velocity", 0.5),
        user_frustration=signals_dict.get("user_frustration", 0.2),
        infra_cost=signals_dict.get("infra_cost", 0.4),
        security_findings=signals_dict.get("security_findings", 0.1),
    )
    qc = build_wormhole_circuit(sig)
    weights = sample_weights_from_circuit(qc)
    ranked = rank_improvements(weights, sig, top_k=6)
    return {
        "weights": weights,
        "recommendations": [
            {"category": c, "score": round(s, 4), "suggestion": text}
            for c, s, text in ranked
        ]
    }

    volatility: float
    dev_velocity: float
    user_frustration: float
    infra_cost: float
    security_findings: float

def _clip01(x): return max(0.0, min(1.0, float(x)))

dev = qml.device("default.qubit", wires=3, shots=2048)

@qml.qnode(dev)
def wormhole_circuit(angles):
    # Encode
    for i, a in enumerate(angles):
        qml.RY(a, wires=i%3)
        qml.RZ(a/2, wires=i%3)

    return qml.sample(wires=[0,1,2])

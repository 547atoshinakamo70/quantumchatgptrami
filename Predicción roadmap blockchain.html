# wormhole_predictor.py
from __future__ import annotations
from dataclasses import dataclass
from typing import Dict, List, Tuple
import math, numpy as np

# --- Qiskit wormhole ---
from qiskit import QuantumCircuit, Aer, execute

CATEGORIES = [
    "latency", "retrieval", "security", "wallet",
    "mining", "zk", "coinjoin", "ui"
]

@dataclass
class Signals:
    # Rellena estos con tus métricas reales
    trend_strength: float     # 0..1 (adopción/crecimiento)
    volatility: float         # 0..1 (ruido/regresiones)
    dev_velocity: float       # 0..1 (cadencia de entrega)
    user_frustration: float   # 0..1 (errores, caídas, quejas)
    infra_cost: float         # 0..1 (coste presión)
    security_findings: float  # 0..1 (hallazgos recientes)

def _clip01(x: float) -> float:
    return max(0.0, min(1.0, float(x)))

def build_wormhole_circuit(sig: Signals) -> QuantumCircuit:
    """
    Codifica señales en rotaciones, aplica SWAP (wormhole) y mide.
    3 qubits → 3 bits → luego expandimos a 8 categorías.
    """
    qc = QuantumCircuit(3, 3)
    # Encode signals in rotations (escala a [0, 2π])
    angles = [
        2*math.pi*_clip01(sig.trend_strength + 0.15*sig.dev_velocity),
        2*math.pi*_clip01(0.5*sig.volatility + 0.5*sig.user_frustration),
        2*math.pi*_clip01(0.6*sig.security_findings + 0.4*sig.infra_cost),
    ]
    for i, a in enumerate(angles):
        qc.ry(a, i)
        qc.rz(a/2, i)

    # “Wormhole”: mezcla y cruza información temporal/causal
    qc.h(0)
    qc.x(1)
    qc.swap(0, 1)
    qc.cx(1, 2)
    qc.h(0)

    # Medida
    qc.measure([0,1,2], [0,1,2])
    return qc

def sample_weights_from_circuit(qc: QuantumCircuit, shots: int = 2048) -> Dict[str, float]:
    backend = Aer.get_backend('qasm_simulator')
    result = execute(qc, backend, shots=shots).result()
    counts = result.get_counts()

    # 3 bits → 8 estados → mapea directamente a 8 categorías
    total = sum(counts.values())
    probs8 = np.zeros(8, dtype=float)
    for bitstring, c in counts.items():
        # bitstring como '101' → índice
        idx = int(bitstring, 2)
        probs8[idx] = c / total

    # Normaliza y suaviza
    probs8 = (probs8 + 1e-6) / (probs8.sum() + 1e-6 * len(probs8))

    weights = {cat: float(probs8[i]) for i, cat in enumerate(CATEGORIES)}
    return weights

# --- Heurística / reglas (puedes cambiar) ---
SUGGESTIONS = {
    "latency":  "Activar streaming + cache de logits + batching; medir p95.",
    "retrieval":"RAG con índice vectorial (FAISS), refresh incremental y guardado por dominios.",
    "security": "Límites/quotas, saneamiento entradas, firmas en RPC, auditoría y fuzz.",
    "wallet":   "Sincronía robusta wallet<->chain, gestión claves y 2FA opcional.",
    "mining":   "Planificador de minería vía HTTP con colas y backoff; métricas de shares.",
    "zk":       "Precompilar circuitos ZK, caché de pruebas, APIs asincrónicas.",
    "coinjoin": "Mejorar selector de monedas + coinjoin programable con políticas.",
    "ui":       "Onboarding simple, plantillas de prompts y diagnósticos visibles."
}

def rank_improvements(weights: Dict[str, float], sig: Signals, top_k: int = 6) -> List[Tuple[str, float, str]]:
    """
    Combina pesos cuánticos + una priorización simple según señales.
    """
    # Boosts por situación:
    boosts = {
        "security": 1.0 + 0.7*_clip01(sig.security_findings),
        "latency":  1.0 + 0.6*_clip01(sig.user_frustration),
        "retrieval":1.0 + 0.4*_clip01(sig.trend_strength),
        "ui":       1.0 + 0.3*_clip01(sig.trend_strength),
        "mining":   1.0 + 0.4*_clip01(sig.dev_velocity),
        "zk":       1.0 + 0.4*_clip01(sig.dev_velocity),
        "coinjoin": 1.0 + 0.3*_clip01(sig.volatility),
        "wallet":   1.0 + 0.4*_clip01(sig.user_frustration),
    }

    scored = []
    for cat in CATEGORIES:
        base = weights.get(cat, 0.0)
        score = base * boosts.get(cat, 1.0)
        scored.append((cat, score, SUGGESTIONS[cat]))

    scored.sort(key=lambda x: x[1], reverse=True)
    return scored[:top_k]

# --- API principal que usaremos desde Flask ---
def predict_roadmap(signals_dict: Dict[str, float]) -> Dict:
    sig = Signals(
        trend_strength=signals_dict.get("trend_strength", 0.5),
        volatility=signals_dict.get("volatility", 0.5),
        dev_velocity=signals_dict.get("dev_velocity", 0.5),
        user_frustration=signals_dict.get("user_frustration", 0.2),
        infra_cost=signals_dict.get("infra_cost", 0.4),
        security_findings=signals_dict.get("security_findings", 0.1),
    )
    qc = build_wormhole_circuit(sig)
    weights = sample_weights_from_circuit(qc)
    ranked = rank_improvements(weights, sig, top_k=6)
    return {
        "weights": weights,
        "recommendations": [
            {"category": c, "score": round(s, 4), "suggestion": text}
            for c, s, text in ranked
        ]
    }

# wormhole_predictor_pl.py (alternativa con PennyLane)
from __future__ import annotations
from dataclasses import dataclass
from typing import Dict, List, Tuple
import math, numpy as np
import pennylane as qml

CATEGORIES = ["latency","retrieval","security","wallet","mining","zk","coinjoin","ui"]

@dataclass
class Signals:
    trend_strength: float
    volatility: float
    dev_velocity: float
    user_frustration: float
    infra_cost: float
    security_findings: float

def _clip01(x): return max(0.0, min(1.0, float(x)))

dev = qml.device("default.qubit", wires=3, shots=2048)

@qml.qnode(dev)
def wormhole_circuit(angles):
    # Encode
    for i, a in enumerate(angles):
        qml.RY(a, wires=i%3)
        qml.RZ(a/2, wires=i%3)
    # “Wormhole”
    qml.Hadamard(wires=0)
    qml.PauliX(wires=1)
    qml.SWAP(wires=[0,1])
    qml.CNOT(wires=[1,2])
    qml.Hadamard(wires=0)
    return qml.sample(wires=[0,1,2])

def predict_roadmap(signals_dict: Dict[str,float]) -> Dict:
    sig = Signals(
        trend_strength=signals_dict.get("trend_strength",0.5),
        volatility=signals_dict.get("volatility",0.5),
        dev_velocity=signals_dict.get("dev_velocity",0.5),
        user_frustration=signals_dict.get("user_frustration",0.2),
        infra_cost=signals_dict.get("infra_cost",0.4),
        security_findings=signals_dict.get("security_findings",0.1),
    )
    angles = np.array([
        2*math.pi*_clip01(sig.trend_strength + 0.15*sig.dev_velocity),
        2*math.pi*_clip01(0.5*sig.volatility + 0.5*sig.user_frustration),
        2*math.pi*_clip01(0.6*sig.security_findings + 0.4*sig.infra_cost),
    ])
    samples = wormhole_circuit(angles)
    # samples: array shape (shots, 3) → bitstrings
    counts = np.zeros(8)
    for s in samples:
        idx = (int(s[0])<<2) | (int(s[1])<<1) | int(s[2])
        counts[idx] += 1
    probs8 = (counts + 1e-6) / (counts.sum() + 8e-6)
    weights = {CATEGORIES[i]: float(probs8[i]) for i in range(8)}

    # Igual que antes (reutiliza SUGGESTIONS y rank_improvements)
    # ... (puedes importar desde el archivo Qiskit para no duplicar)
    from wormhole_predictor import rank_improvements
    ranked = rank_improvements(weights, sig, top_k=6)
    return {
        "weights": weights,
        "recommendations": [
            {"category": c, "score": round(s,4), "suggestion": text}
            for c,s,text in ranked
        ]
    }

Ejemplos de mejoras que te propondrá (y por qué)

  • latency ↑ si user_frustration o volatility suben: activa streaming, cache de logits, batching de requests, y usa context windows más cortas con RAG por chunks (mejora p95 y coste).

  • retrieval (RAG) ↑ si trend_strength sube: índice FAISS/HNSW por dominios (blockchain, wallet, minería), refresco incremental y re-rank ligero para precisión.

  • security ↑ si security_findings > 0: firmas RPC, validaciones en servidor, sandboxes de prompts, fuzz de endpoints, testing de prompts con políticas.

  • wallet/coinjoin/mining según tu enfoque:

    • Wallet: autenticación email+contraseña y derivación de claves segura (lo que querías), 2FA opcional, cuentas “observadas”.

    • Coin selection/Join: BnB/Branch-and-Bound + políticas, coinjoin programable por metas (privacidad/costo).

    • Minería vía HTTP: colas, backoff exponencial, métricas por worker, y planificador sencillo para jobs.

def create_app():

        # App privada: si hay API_KEY, exige encabezado X-API-Key

            abort(401)
    return app

app = create_app()

4) GHCR (imagen privada) + CI

Crea .github/workflows/build.yml:

yaml
name: build-and-push
on:
  push:
    branches: [ "main" ]
permissions:
  contents: read
  packages: write

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Log in to GHCR
        uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Extract metadata
        id: meta
        uses: docker/metadata-action@v5
        with:
          images: ghcr.io/${{ github.repository }}
          tags: |
            type=raw,value=latest
            type=ref,event=branch
            type=sha

      - name: Build and push
        uses: docker/build-push-action@v6
        with:
          context: .
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}

Después del primer push a main, tendrás tu imagen privada en GHCR (ghcr.io/OWNER/REPO:latest).

5) Despliegue privado en una máquina (opciones)

Opción A — Solo local (máquina personal):

Opción B — VPS/servidor pero sin exponer a Internet:

  • Mantén ports: "127.0.0.1:8000:8000" y accede con túnel SSH:

    bash
    ssh -N -L 8000:127.0.0.1:8000 user@tu-servidor

  • Alternativa: Tailscale/WireGuard para acceso privado en red malla.

Opción C — Exponer con TLS + Auth (si algún día lo necesitas):

  • Pon un proxy (Caddy/Nginx) delante con HTTPS y auth básica o mTLS.

  • De todos modos conserva la API key; añade rate limiting al proxy.
